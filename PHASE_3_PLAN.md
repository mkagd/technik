function calculateScore(tech, visit) {

  // 1. Specjalizacja (+30)

  }
  
  // 3. Odleg≈Ço≈õƒá (-1 za ka≈ºdy km)
  const distance = calculateDistance(tech.location, visit.address);
  score -= distance;
  reasons.push('Odleg≈Ço≈õƒá: ' + distance + ' km');
  
  // 4. ObciƒÖ≈ºenie dzisiaj (-5 za ka≈ºdƒÖ wizytƒô)
